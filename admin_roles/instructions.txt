kubectl apply -f admin_sa_token_secret.yaml 
kubectl get configmap -n kube-system kube-root-ca.crt -o jsonpath='{.data.ca\.crt}' > ca.crt
kubectl config set-cluster kubernetes --server=https://10.10.3.10:6443 --certificate-authority=./ca.crt --embed-certs=true --kubeconfig=admin-kubeconfig
kubectl config set-context admin-context --cluster=kubernetes --user=admin-sa --namespace=kube-system --kubeconfig=admin-kubeconfig
kubectl config use-context admin-context --kubeconfig=admin-kubeconfig

kubectl describe secret/admin-sa-token -n kube-system
